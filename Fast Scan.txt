#include<bits/stdc++.h>
using namespace std;

void fastscan(int &n) 
{  
  bool neg = false; 
  register int c; 

  n = 0; 
  c = getchar(); 
  if(c=='-') 
  {  
    neg = true;  
    c = getchar(); 
  }  
  for(; (c>47 && c<58); c=getchar()) 
     n = n *10 + c - 48; 
 
  if(neg) 
    n *= -1; 
} 

// Function Call 
int main() 
{ 
  int number; 
  fastscan(number); 

  return 0; 
}
